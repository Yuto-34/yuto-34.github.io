	m\dv{x}{t} &= \lambda x \\
	\int\dfrac{\mathrm{d}x}{x} &= \int \dfrac{\lambda }{m} \mathrm{d}t \\
	\log x &= \dfrac{\lambda }{m}t + c \\
	x &= \exp\qty[\dfrac{\lambda }{m}t + c] \\
	&= e^c\exp\qty[\dfrac{\lambda }{m}t] \\
	&= C\exp\qty[\dfrac{\lambda }{m}t] \qq{($\because e^c = \mathrm{const}. = C$)}
\end{align*}
